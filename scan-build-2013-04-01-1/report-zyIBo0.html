455void SimpleIncNodeInserter::constructDual(const Graph &G,
456 const CombinatorialEmbedding &E,
457 //node vCopy,
458 bool forbidCrossings)
459{
460 m_dual.clear();
461
462 // insert a node in the dual graph for each face in E
463 face f;
464 forall_faces(f,E)for((f)=(E).firstFace(); (f); (f)=(f)->succ())
465 m_nodeOf[f] = m_dual.newNode();
466
467 // Insert an edge into the dual graph for each adjacency entry in E.
468 // The edges are directed from the left face to the right face.
469 node v;
470 forall_nodes(v,G)for((v)=(G).firstNode(); (v); (v)=(v)->succ())
471 {
472 adjEntry adj;
473 forall_adj(adj,v)for((adj)=(v)->firstAdj(); (adj); (adj)=(adj)->succ())
474 {
475 node vLeft = m_nodeOf[E.leftFace (adj)];
476 node vRight = m_nodeOf[E.rightFace(adj)];
477
478 edge e = m_dual.newEdge(vLeft,vRight);
479 m_primalAdj[e] = adj;
480
481 // mark dual edges corresponding to generalizations
482 if (forbidCrossings)
483 if (m_planRep->typeOf(adj->theEdge()) == Graph::generalization)
484 m_primalIsGen[e] = true;
485 }
486 }
487
488 // Augment the dual graph by two new vertices. These are used temporarily
489 // when searching for a shortest path in the dual graph.
490 m_vS = m_dual.newNode();
491 m_vT = m_dual.newNode();
492
493}//constructdual

496//---------------------------------------------------------
497// finds a shortest path in the dual graph augmented by s and t (represented
498// by m_vS and m_vT); returns list of crossed adjacency entries (corresponding
499// to used edges in the dual) in crossed.
500// Used to insert edges with crossings adjacent to insertion node v
501void SimpleIncNodeInserter::findShortestPath(
502 const CombinatorialEmbedding &E,
503 node s,
504 node t,
505 Graph::EdgeType eType,
506 SList<adjEntry> &crossed)
507{
508 OGDF_ASSERT(s != t);
509//if (!isConnected(m_dual)) cout<<"Not connected\n"<<flush;
510
511 NodeArray<edge> spPred(m_dual,0);
512 QueuePure<edge> queue;
513 int oldIdCount = m_dual.maxEdgeIndex();
514
515 // augment dual by edges from s to all adjacent faces of s ...
516 adjEntry adj;
517 forall_adj(adj,s)for((adj)=(s)->firstAdj(); (adj); (adj)=(adj)->succ())
518 {
519 // starting edges of bfs-search are all edges leaving s
520 edge eDual = m_dual.newEdge(m_vS, m_nodeOf[E.rightFace(adj)]);
521 m_primalAdj[eDual] = adj;
522 queue.append(eDual);
523 }
524
525 // ... and from all adjacent faces of t to t
526 forall_adj(adj,t)for((adj)=(t)->firstAdj(); (adj); (adj)=(adj)->succ()) {
527 edge eDual = m_dual.newEdge(m_nodeOf[E.rightFace(adj)], m_vT);
528 m_primalAdj[eDual] = adj;
529 }
530
531 // actual search (using bfs on directed dual)
532 for( ; ; )
533 {
534 // next candidate edge
535 edge eCand = queue.pop();
536 node v = eCand->target();
537
538 // leads to an unvisited node?
539 if (spPred[v] == 0)
540 {
541 // yes, then we set v's predecessor in search tree
542 spPred[v] = eCand;
543
544 // have we reached t ...
545 if (v == m_vT)
546 {
547 // ... then search is done.
548 // constructed list of used edges (translated to crossed
549 // adjacency entries in PG) from t back to s (including first
550 // and last!)
551
552 do {
553 edge eDual = spPred[v];
554 crossed.pushFront(m_primalAdj[eDual]);
555 v = eDual->source();
556 } while(v != m_vS);
557
558 break;
559 }
560
561 // append next candidate edges to queue
562 // (all edges leaving v)
563 edge e;
564 forall_adj_edges(e,v)for(ogdf::adjEntry ogdf_loop_var=(v)->firstAdj(); ogdf::test_forall_adj_edges
(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var->succ()) {
565 if (v == e->source() &&
566 (eType != Graph::generalization || m_primalIsGen[e] == false))
567 {
568 queue.append(e);
569 }
570 }
571 }
572 }
573
574
575 // remove augmented edges again
576 while ((adj = m_vS->firstAdj()) != 0)
577 m_dual.delEdge(adj->theEdge());
578
579 while ((adj = m_vT->firstAdj()) != 0)
580 m_dual.delEdge(adj->theEdge());
581
582 m_dual.resetEdgeIdCount(oldIdCount);
583}//findshortestpath

586// inserts edge e according to insertion path crossed.
587// updates embedding and dual graph

589void SimpleIncNodeInserter::insertEdge(
590 CombinatorialEmbedding &E,
591 edge eOrig,
592 const SList<adjEntry> &crossed,
593 bool forbidCrossingGens)
594{
595 // remove dual nodes on insertion path
596 SListConstIterator<adjEntry> it;
597 for(it = crossed.begin(); it != crossed.rbegin(); ++it) {
598 m_dual.delNode(m_nodeOf[E.rightFace(*it)]);
599 }
600
601 // update primal
602 m_planRep->insertEdgePathEmbedded(eOrig,E,crossed);
603
604 // insert new face nodes into dual
605 const List<edge> &path = m_planRep->chain(eOrig);
606 ListConstIterator<edge> itEdge;
607 for(itEdge = path.begin(); itEdge.valid(); ++itEdge)
608 {
609 adjEntry adj = (*itEdge)->adjSource();
610 m_nodeOf[E.leftFace (adj)] = m_dual.newNode();
611 m_nodeOf[E.rightFace(adj)] = m_dual.newNode();
612 }
613
614 // insert new edges into dual
615 for(itEdge = path.begin(); itEdge.valid(); ++itEdge)
616 {
617 adjEntry adjSrc = (*itEdge)->adjSource();
618 face f = E.rightFace(adjSrc); // face to the right of adj in loop
619 node vRight = m_nodeOf[f];
620
621 adjEntry adj1 = f->firstAdj(), adj = adj1;
622 do {
623 node vLeft = m_nodeOf[E.leftFace(adj)];
624
625 edge eLR = m_dual.newEdge(vLeft,vRight);
626 m_primalAdj[eLR] = adj;
627
628 edge eRL = m_dual.newEdge(vRight,vLeft);
629 m_primalAdj[eRL] = adj->twin();
630
631 if(forbidCrossingGens &&
632 m_planRep->typeOf(adj->theEdge()) == Graph::generalization)
633 {
634 m_primalIsGen[eLR] = m_primalIsGen[eRL] = true;
635 }
636 }
637 while((adj = adj->faceCycleSucc()) != adj1);
638
639 // the other face adjacent to *itEdge ...
640 f = E.rightFace(adjSrc->twin());
641 vRight = m_nodeOf[f];
642
643 adj1 = f->firstAdj();
644 adj = adj1;
645 do {
646 node vLeft = m_nodeOf[E.leftFace(adj)];
647
648 edge eLR = m_dual.newEdge(vLeft,vRight);
649 m_primalAdj[eLR] = adj;
650
651 edge eRL = m_dual.newEdge(vRight,vLeft);
652 m_primalAdj[eRL] = adj->twin();
653
654 if(forbidCrossingGens &&
655 m_planRep->typeOf(adj->theEdge()) == Graph::generalization)
656 {
657 m_primalIsGen[eLR] = m_primalIsGen[eRL] = true;
658 }
659 }
660 while((adj = adj->faceCycleSucc()) != adj1);
661 }
662}//insertedge
