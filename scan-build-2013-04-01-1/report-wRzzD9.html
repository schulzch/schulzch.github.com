118// Calculates the number of overlapping graphs in one edge and gives them
119// a color calculated from the choosen colorscheme
120String SimDrawColorizer::SimDrawColorScheme::getColor(int subGraphBits, int numberOfGraphs)
121{
122 String color = "#", s;
123 int r = 0x00, g = 0x00, b = 0x00;
124 int numberOfGraphsInEdge = 0;
125 Array<bool> graphs(numberOfGraphs); //Ueberlagerungen von Graphen bei dieser Kante
126
127 /* Loest den Integerwert SubGraphBits in die einzelnen Bits auf und
128 findet somit heraus, welche Graphen sich in dieser Kante ueberlagern */
129 for (int i = 0; i < numberOfGraphs; i++)
130 {
131 graphs[i] = 0;
132 if((subGraphBits & (1 << i)) != 0)
133 graphs[i]=1;
134 }
135
136 /* Bestimmt den Mittelwert der Farben der uebereinanderliegenden Graphen */
137 for (int i = 0; i < numberOfGraphs; i++)
138 {
139 if (graphs[i] == 1)
140 {
141 r += red[i];
142 g += green[i];
143 b += blue[i];
144 numberOfGraphsInEdge++;
145 }
146 }
147 if (numberOfGraphsInEdge == numberOfGraphs)
148 {
149 r = 0x00; // Kanten werden schwarz eingefaerbt
150 g = 0x00; // wenn sie zu allen Graphen gehoeren
151 b = 0x00;
152 }
153 else
154 {
155 OGDF_ASSERT(numberOfGraphsInEdge > 0);
156 r /= numberOfGraphsInEdge;
157 g /= numberOfGraphsInEdge;
158 b /= numberOfGraphsInEdge;
159 }
160
161 /* Setzt die einzelnen Farben zu eine Hex Farbcode zusammen */
162 s.sprintf("%x",r);
163 if (s.length() == 1) color += "0";
164 color += s;
165 s.sprintf("%x",g);
166 if (s.length() == 1) color += "0";
167 color += s;
168 s.sprintf("%x",b);
169 if (s.length() == 1) color += "0";
170 color += s;
171
172 return color;
173
174} // end getColor
