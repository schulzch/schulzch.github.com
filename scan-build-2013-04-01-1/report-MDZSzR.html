156double atan2ex(double y, double x)
157{
158 double angle = atan2(y, x);
159
160 if (x == 0)
161 {
162 if (y >= 0) {
163 angle = 0.5 * Math::pi;
164 } else {
165 angle = 1.5 * Math::pi;
166 }
167 }
168
169 if (y == 0)
170 {
171 if (x >= 0)
172 {
173 angle = 0.0;
174 } else {
175 angle = Math::pi;
176 }
177 }
