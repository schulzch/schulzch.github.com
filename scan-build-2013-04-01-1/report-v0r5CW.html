676 x[actNode] = x[mdl[0]] - mDist[mdl[0]] + mDist[actNode];
677 actNode = mdl[1];
678 while(++actNode <= right &&
679 x[mdl[1]] - x[actNode] > mDist[mdl[1]] - mDist[actNode])
680 x[actNode] = x[mdl[1]] - mDist[mdl[1]] + mDist[actNode];
681 }
682}

686/*************************************************************************
687 moveLongEdge
688**************************************************************************
689
690The function moveLongEdge is used for postprocessing the layout.
691If the two nonvirtual ndoes of the long edge are both to the left (right) of
692the virtual nodes, the function moveLongEdge tries to reduce the length of the
693two outermost segments by moving the virtual nodes simultaneously as far as
694possible to the left (right). If both non virtual nodes are on different sides
695of the virtual nodes, moveLongEdge tries to remove one of the edge bends by
696moving the virtual nodes.
697
698
699If there exists a conflict with another long edge on the left (right) side of
700the current long edge, the function moveLongEdge is first applied recursively
701to this long edge.
702
703actNode : a representative node of the long edge
704dir : is -1 if it is preferred to move the long edge to the left,
705 1 if it is preferred to move the long edge to the right,
706 0 if there is no preference
707marked : array for all nodes. Stores for every node, whether moveLongEdge
708 has already been applied to it.
709
710*/
711
712void FastHierarchyLayout::moveLongEdge(int actNode,
713 int dir,
714 bool *marked)
715{
716 ListConstIterator<int> _it = 0;
717
718 if(!marked[actNode]&&virt[actNode]) {
719 // if actNode belongs to a long edge and has not been moved yet.
720 int next;
721 // mark all virtual nodes of the long edge
722 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++) {
723 next = *_it;
724 marked[(*_it)]=1;
725 }
726
727 // first non virtual node of long edge
728 int fst=adj[0][longEdge[actNode]->front()].front();
729
730 // second non virtual node of long edge
731 int lst=adj[1][longEdge[actNode]->back()].front();
732
733 // Contains an order of the two positions of the nonvirtual nodes of
734 // the long edge. The function moveLongEdge first tries to place the
735 // long edge onto the first position If not successful, moveLongEdge
736 // tries to place the long edge onto the second position
737 List<double> toTest;
738
739 if(dir < 0) {
740 toTest.pushBack(x[fst] < x[lst] ? x[fst] : x[lst]);
741 toTest.pushBack(x[fst] < x[lst] ? x[lst] : x[fst]);
742 }
743 else if(dir > 0) {
744 toTest.pushBack(x[fst] < x[lst] ? x[lst] : x[fst]);
745 toTest.pushBack(x[fst] < x[lst] ? x[fst] : x[lst]);
746 }
747 else {
748 toTest.pushBack(x[fst]);
749 toTest.pushBack(x[lst]);
750 }
751 double xFirst; // stores the first and most preferred position
752 xFirst = toTest.front();
753
754 double xOpt; // stores the preferred position
755
756 bool done = false;
757 while(!done && !toTest.empty()) {
758 xOpt = toTest.front(); // Best position that can be reached by moving
759 // the long edge
760 toTest.popFront();
761 done = 1;
762 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++) {
763 // for all virtual nodes on the long edge
764 next = *_it;
765
766 // Try moving to the left
767 if(!isFirst(next)) {
768 // next does have a left sibling
769 if(xOpt - x[next-1] < totalB[next] - totalB[next-1])
770 // there is a conflict moving next to the position xOpt.
771 moveLongEdge(next - 1,-1,marked);
772
773 // done = 0 if minimal distances cannot be repsected.
774 done = done &&
775 xOpt - x[next-1] >=
776 totalB[next] - totalB[next-1] - ALLOW.00001;
777 }
778
779 // Try moving to the right
780 if(!isLast(next)) {
781 // next does have a right sibling
782 if(xOpt - x[next+1] > totalB[next] - totalB[next+1])
783 // there is a conflict moving next to the position xOpt.
784 moveLongEdge(next + 1,1,marked);
785
786 // done = 0 if minimal distances cannot be respected.
787 done = done &&
788 xOpt - x[next+1] <=
789 totalB[next] - totalB[next+1] + ALLOW.00001;
790 }
791 }
792 }
793 if(!done) {
794 // moveLongEdge was not able to move the virtual nodes to one of
795 // the two positions of the nonvirtual nodes. It now tries to
796 // approximate the most preferred position.
797 xOpt = xFirst;
798 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++) {
799 next = *_it;
800 if(!isFirst(next))
801 incrTo(xOpt,x[next-1]+totalB[next]-totalB[next-1]);
802 if(!isLast(next))
803 decrTo(xOpt,x[next+1]+totalB[next]-totalB[next+1]);
804 }
805 }
806 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++) {
807 next = *_it;
808 x[(*_it)] = xOpt;
809 }
810 }
811}

814/*************************************************************************
815 straightenEdge
816**************************************************************************
817
818The function straightenEdge is applied to long edges with exactly one virtual
819node and tries to remove a bend at the position of the virtual node, by
820straightening the edge.
821
822actNode : the virtual representative node of the long edge
823marked : array for all nodes. Stores for every node, whether straightenEdge
824 has already been applied to it.
825
826If there exists a conflict with a direct sibling to the left (right) side of
827the current node, the function straightenEdge is first applied recursively to
828this node.
829
830*/
831
832void FastHierarchyLayout::straightenEdge(int actNode,bool *marked)
833{
834 if(!marked[actNode] && // breadth[actNode] < ALLOW &&
835 adj[0][actNode].size() == 1 &&
836 adj[1][actNode].size() == 1 &&
837 longEdge[actNode]->size() < 2)
838 {
839 marked[actNode]=1;
840
841 int fst=adj[0][actNode].front();
842 int lst=adj[1][actNode].front();
843
844 // Get optimal position for actNode
845 double xOpt=x[fst] + (x[lst] - x[fst]) *
846 (y[layer[actNode]] - y[layer[fst]]) /
847 (y[layer[lst]] - y[layer[fst]]);
848
849
850 if(!isFirst(actNode)) {
851 // actNode does have a left sibling
852 if(xOpt - x[actNode-1] < totalB[actNode] - totalB[actNode-1] - ALLOW.00001)
853 // there is a conflict with the position of the direct left sibling
854 // Recursively call straightenEdge.
855 straightenEdge(actNode - 1,marked);
856 if(xOpt - x[actNode - 1]<totalB[actNode] - totalB[actNode-1]-ALLOW.00001)
857 return;
858 }
859 if(!isLast(actNode)) {
860 // actNode does have a right sibling
861 if(x[actNode+1] - xOpt < totalB[actNode+1] - totalB[actNode] - ALLOW.00001)
862 // there is a conflict with the position of the direct left sibling
863 // Recursively call straightenEdge.
864 straightenEdge(actNode + 1,marked);
865 if(x[actNode+1] - xOpt < totalB[actNode+1] - totalB[actNode] - ALLOW.00001)
866 return;
867 }
868 x[actNode] = xOpt;
869 }
870}

874/*************************************************************************
875 findPlacement
876**************************************************************************
877
878The function findPlacement computes the layout of an embedded layered graph.
879
880*/
881
882void FastHierarchyLayout::findPlacement()
883{
884 int actNode,next,last,actLayer,dir,leftBnd,leftNxt,rightNxt;
885 bool *marked=new bool[n];
886 ListConstIterator<int> _it = 0;
887
888 // Replace all virtual nodes in an edge traversing only one layer by a
889 // nonvirtual node.
890 forallnodesfor(actNode=0;actNode<n;actNode++) if(virt[actNode] &&
891 !virt[adj[0][actNode].front()] &&
892 !virt[adj[1][actNode].front()])
893 virt[actNode] = 0;
894
895 // Compute minimal distances between nodes (totalB)
896
897 totalB = new double[n];
898 double toAdd;
899 forallnodesfor(actNode=0;actNode<n;actNode++) {
900 if (!actNode ||
901 layer[actNode-1] < layer[actNode])
902 totalB[actNode]=0;
903 else {
904 toAdd = breadth[actNode-1] / 2 + breadth[actNode] / 2;
905 // Enlarge the minimal Distance for nodes with a lot of neighbours.
906 incrTo(toAdd,m_minNodeDist / 3 * (double)(adj[0][actNode-1].size() + adj[0][actNode].size()));
907 incrTo(toAdd,m_minNodeDist / 3 * (double)(adj[1][actNode-1].size() + adj[1][actNode].size()));
908 // distances are computed such that they are placed on a grid based on minNodeDist.
909 toAdd = m_minNodeDist *(int)(toAdd/m_minNodeDist + 1 - ALLOW.00001);
910 toAdd += m_minNodeDist;
911 totalB[actNode] = totalB[actNode-1] + toAdd;
912 }
913 }
914
915 // Remove crossings of long edges
916 // Applied for long edges that cross each other in inner segments.
917
918 List<int> *newEdge,*oldEdge;
919 int down,spl;
920
921
922 // For every two consecutive layers l and l+1 , traverse the inner edge
923 // segments between the two layers according to the order of virtual nodes
924 // on layer l. We consider consecutive pairs of inner edges segments
925 // from left to right. If two inner edge segments cross, we split the
926 // right long edge between the two layers by parting the corresponding list
927 // into two sublists.
928 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
929 last=-1;
930 forallnodesonlayerfor(actNode=first[actLayer];actNode<first[actLayer+1];actNode
++) {
931 if(virt[actNode]) {
932 down=adj[1][actNode].front();
933 if(virt[down]) {
934 if(last!=-1&&last>down) {
935 oldEdge = longEdge[actNode];
936 spl = actLayer - layer[oldEdge->front()] + 1;
937 newEdge = OGDF_NEWnew List<int>;
938 oldEdge->split(oldEdge->get(spl),(*newEdge),
939 (*oldEdge));
940 for (_it = (*newEdge).begin(); _it.valid(); _it++) {
941 next = *_it;
942 longEdge[next] = newEdge;
943 }
944 } else
945 last = down;
946 }
947 }
948 }
949 }
950
951 // Place long edges
952
953 int blockCount;
954 bool exD;
955 int *block = new int[n];
956 double *pos = new double[n];
957 List<int> *blockNodes;
958 kList neighbours;
959 double dist;
960
961 forallnodesfor(actNode=0;actNode<n;actNode++)
962 x[actNode] = 0;
963 for(dir = 1; dir >= -1;dir -= 2) {
964 // for dir = 1, move long edges as far as possible to the left
965 // for dir = -1, move long edges as far as possible to the right
966
967
968 // Partition the graph into blocks according to the technical report.
969 blockCount = 0;
970 forallnodesfor(actNode=0;actNode<n;actNode++)
971 block[actNode] = marked[actNode] = 0;
972 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
973 actNode = (dir == 1 ? first[actLayer] : first[actLayer+1] - 1);
974 if(!block[actNode]) {
975 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++){
976 next = *_it;
977 block[next] = blockCount;
978 }
979 blockCount++;
980 }
981 actNode += dir;
982 while(actNode >= first[actLayer] && actNode < first[actLayer+1]) {
983 if(!block[actNode]) {
984 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++){
985 next = *_it;
986 block[next] = block[actNode-dir];
987 }
988 }
989 actNode += dir;
990 }
991 }
992
993 // Store the nodes of every block in a separate list
994 blockNodes = new List<int>[n];
995 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
996 for(actNode = (dir == 1 ? first[actLayer] : first[actLayer+1]-1);
997 actNode >= first[actLayer] && actNode < first[actLayer+1];
998 actNode += dir)
999 {
1000 blockNodes[block[actNode]].pushBack(actNode);
1001 }
1002 }
1003
1004 for(int i = 0; i < blockCount;i++) {
1005 // for every block
1006 exD = 0;
1007 dist = 0;
1008 for (_it = blockNodes[i].begin(); _it.valid(); _it++) {
1009 // for every node of the block apply sortLongEdges
1010 actNode = *_it;
1011 sortLongEdges(actNode,dir,pos,exD,dist,block,marked);
1012 }
1013 if(!exD) {
1014 // The currently examined block does not share its layers with
1015 // another block that has been placed before. Thus the block
1016 // can be freely moved. We compute a position for the block by
1017 // minimizing the total edge length to neighbours of blocks
1018 // that have already been placed. dist has not been computed by
1019 // sortLongEdges and is now set to the optimal value.
1020 actLayer = layer[blockNodes[i].front()];
1021 forallnodesonlayerfor(actNode=first[actLayer];actNode<first[actLayer+1];actNode
++) {
1022 for (_it = adj[0][actNode].begin(); _it.valid(); _it++) {
1023 next = *_it;
1024 neighbours.add(0,pos[next] - pos[actNode]);
1025 }
1026 }
1027 if(!neighbours.empty()) {
1028 cmpWithKey cmp;
1029 neighbours.quicksort(cmp);
1030 dist =- dir*neighbours.median();
1031 neighbours.clear();
1032 }
1033 }
1034 // If exD is true, dist has been computed by sortLongEdges.
1035
1036 // Move the nodes of the block to their positions.
1037 for (_it = blockNodes[i].begin(); _it.valid(); _it++) {
1038 actNode = *_it;
1039 pos[actNode] -= dir * dist;
1040 }
1041 }
1042
1043 // Compute for every node the average of the two positions of the left
1044 // and right placement.
1045 forallnodesfor(actNode=0;actNode<n;actNode++)
1046 x[actNode] += pos[actNode] / 2;
1047
1048 delete[] blockNodes;
1049 }
1050
1051 delete[] block;
1052 delete[] pos;
1053
1054 // Place nonvirtual nodes
1055
1056 // Stores for every internal sequence of in
1057 // wihch traversal the sequence has to be placed.
1058 // An internal sequence is represeted by its left virtual sibling. Values:
1059 // -1 : is not yet clear
1060 // 0 : top down traversal
1061 // 1 : bottom up traversal
1062 int *nodeDir = new int[n];
1063
1064 forallnodesfor(actNode=0;actNode<n;actNode++)
1065 nodeDir[actNode] = -1;
1066
1067
1068
1069 // Initialization of marked.
1070 // marked is used to indicate the placement of a sequence.
1071 forallnodesfor(actNode=0;actNode<n;actNode++) {
1072 if(virt[actNode]) {
1073 next = actNode + 1;
1074 while(sameLayer(next,actNode) && !virt[next])
1075 next++;
1076
1077 // If actNode is representative of a fixed inner sequence, the
1078 // sequence is already placed and marked is set
1079 marked[actNode] = sameLayer(next,actNode) &&
1080 x[next] - x[actNode] < totalB[next] - totalB[actNode] + ALLOW.00001;
1081
1082 } else
1083 marked[actNode] = 0;
1084 }
1085
1086 forallnodesfor(actNode=0;actNode<n;actNode++) {
1087 if(marked[actNode] && nodeDir[actNode] == -1) {
1088 // for every fixed sequence, traverse the long edge belonging to
1089 // the representative actnode
1090 for (_it = (*longEdge[actNode]).begin(); _it.valid(); _it++) {
1091 // for every virtual node next of the long edge mark the
1092 // corrsponding sequence as to be placed in the upward
1093 // traversal if next is above of actnode and in the downward
1094 // traversal if next is below actNode.
1095 next = *_it;
1096 if(next != actNode)
1097 nodeDir[next] = next < actNode;
1098 }
1099 }
1100 }
1101
1102 // mDist is equal to totalB during the first traversal. After the first
1103 // traversal it is set to the positions computed in the first traversal.
1104 // It is only modified for external sequences.
1105 mDist = new double[n];
1106 // Die folgende forallnodes Schleife wurde f?r den Vorschlag von
1107 // Christoph (siehe unten) entfernt, da ?berfl?ssig
1108 forallnodesfor(actNode=0;actNode<n;actNode++)
1109 mDist[actNode] = totalB[actNode];
1110
1111 for(dir = 0;dir < 2;dir++) {
1112 // for every traversal
1113
1114 for(actLayer = dir ? k-1 : 0; 0 <= actLayer && actLayer < k;
1115 actLayer += dir ? -1 : 1)
1116 {
1117 // for every layer (if dir = 0 top down)
1118
1119/* // NEU : ?nderungen vorgeschlagen von Christoph
1120 // f?hren aber dazu, dass sich Knoten zu Nahe kommen
1121 // k?nnen (n?her als minDist) (Carsten)
1122 for(int i1 = first[actLayer]; i1 < first[actLayer+1]; i1++)
1123 mDist[i1] = x[i1];
1124
1125 leftBnd=-1;
1126 forallnodesonlayer {
1127 if(virt[actNode] && virt[adj[dir][actNode].front()]) {
1128 x[actNode] = x[adj[dir][actNode].front()];
1129
1130 if(leftBnd == -1)
1131 placeNodes(-1,actNode,first[actLayer],actNode-1,dir);
1132 else
1133 placeNodes(leftBnd,actNode,leftBnd+1,actNode-1,dir);
1134
1135 leftBnd = actNode;
1136 }
1137 }
1138
1139 if(leftBnd == -1)
1140 placeNodes(-1,-1,first[actLayer],first[actLayer+1]-1,dir);
1141 else
1142 placeNodes(leftBnd,-1,leftBnd+1,first[actLayer+1]-1,dir);
1143
1144 for(int i1 = first[actLayer]; i1 < first[actLayer+1]; i1++)
1145 mDist[i1] = totalB[i1];
1146 // ENDE NEU*/
1147
1148 leftBnd=-1;
1149 forallnodesonlayerfor(actNode=first[actLayer];actNode<first[actLayer+1];actNode
++) {
1150 if(virt[actNode]) {
1151 // For every sequence, bounded by leftBnd and actNode.
1152 if(leftBnd == -1) {
1153 // leftBnd is not a node. The sequence is external. Place it.
1154 placeNodes(-1,actNode,first[actLayer],actNode - 1,dir);
1155 for(next = first[actLayer]; next < actNode; next++)
1156 mDist[next] = mDist[actNode] - x[actNode] + x[next];
1157 }
1158 else if(nodeDir[leftBnd] != !dir) { // nodeDir[leftBnd] == dir || nodeDir[leftBnd] == -1
1159 // internal sequence
1160
1161 if(!marked[leftBnd])
1162 // sequence has not been placed yet. Place it.
1163 placeNodes(leftBnd,actNode,leftBnd + 1,
1164 actNode - 1,dir);
1165
1166 // Adjust nodeDir for the next layer.
1167 leftNxt = adj[!dir][leftBnd].front();
1168 rightNxt = adj[!dir][actNode].front();
1169 if(virt[leftNxt] && virt[rightNxt])
1170 for(next = leftNxt + 1;next < rightNxt; next++)
1171 nodeDir[next] = dir;
1172 }
1173 leftBnd = actNode;
1174 }
1175 }
1176 if(leftBnd == -1) {
1177 // No virtual node in the complete layer. Place it.
1178 placeNodes(-1,-1,first[actLayer],first[actLayer+1] - 1,dir);
1179 for (next = first[actLayer]; next < first[actLayer+1]; next++)
1180 mDist[next] = x[next];
1181 }
1182 else {
1183 // External sequence to the right. Place it.
1184 placeNodes(leftBnd,-1,leftBnd + 1,first[actLayer+1] - 1,dir);
1185 for (next = first[actLayer+1] - 1;next > leftBnd; next--)
1186 mDist[next] = mDist[leftBnd] - x[leftBnd] + x[next];
1187 }
1188 }
1189 }
1190 delete[] nodeDir;
1191 delete[] mDist;
1192
1193 // Apply moveLongEdge to every long edge.
1194 forallnodesfor(actNode=0;actNode<n;actNode++)
1195 marked[actNode] = 0;
1196 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
1197 for (actNode = (first[actLayer] + first[actLayer + 1]) / 2;
1198 actNode < first[actLayer+1]; actNode++)
1199 {
1200 moveLongEdge(actNode,0,marked);
1201 }
1202 for (actNode = (first[actLayer] + first[actLayer+1]) / 2 - 1;
1203 actNode >= first[actLayer]; actNode--)
1204 {
1205 moveLongEdge(actNode,0,marked);
1206 }
1207 }
1208
1209 // Compute ordinates for the layers and boxY
1210
1211 double boxY = k ? height[0] / 2 : 0; // y-value for the bounding box.
1212 double minD;
1213
1214 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
1215 y[actLayer] = boxY;
1216 minD = m_minLayerDist;
1217 if(!m_fixedLayerDist) {
1218 forallnodesonlayerfor(actNode=first[actLayer];actNode<first[actLayer+1];actNode
++) {
1219 // adjust the distance of the layer to the longest edge
1220 for (_it = adj[1][actNode].begin(); _it.valid(); _it++) {
1221 next = *_it;
1222 incrTo(minD,(x[next] - x[actNode]) / 3);
1223 incrTo(minD,(x[actNode] - x[next]) / 3);
1224 }
1225 }
1226 decrTo(minD,10 * m_minLayerDist);
1227 }
1228 boxY += height[actLayer] / 2;
1229 if(actLayer < k - 1)
1230 boxY += minD + height[actLayer+1] / 2;
1231 }
1232
1233 // Apply straightenEdge to every long edge with one virtual node.
1234
1235 forallnodesfor(actNode=0;actNode<n;actNode++)
1236 marked[actNode] = 0;
1237 foralllayersfor(actLayer=0;actLayer<k;actLayer++) {
1238 for (actNode = (first[actLayer] + first[actLayer+1]) / 2;
1239 actNode < first[actLayer+1]; actNode++)
1240 {
1241 straightenEdge(actNode,marked);
1242 }
1243 for (actNode = (first[actLayer] + first[actLayer+1]) / 2 - 1;
1244 actNode >= first[actLayer]; actNode--)
1245 {
1246 straightenEdge(actNode,marked);
1247 }
1248 }
1249 delete[] marked;
1250 delete[] totalB;
1251}
