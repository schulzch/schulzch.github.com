87void RadialTreeLayout::call(GraphAttributes &AG)
88{
89 const Graph &tree = AG.constGraph();
90 if(tree.numberOfNodes() == 0) return;
91
92 if (!isTree(tree))
93 OGDF_THROW_PARAM(PreconditionViolatedException, pvcForest)throw PreconditionViolatedException ( pvcForest );
94
95 OGDF_ASSERT(m_levelDistance > 0);
96
97 // determine root of tree (m_root)
98 FindRoot(tree);
99
100 // compute m_level[v], m_parent[v], m_leaves[v], m_numLevels
101 ComputeLevels(tree);
102
103 // computes diameter of each node
104 ComputeDiameters(AG);
105
106 // computes m_angle[v] and m_wedge[v]
107 ComputeAngles(tree);
108
109 // computes final coordinates of nodes
110 ComputeCoordinates(AG);
111}

114void RadialTreeLayout::FindRoot(const Graph &G)
115{
116 node v;
117
118 switch(m_selectRoot) {
119 case rootIsSource:
120 forall_nodes(v,G)for((v)=(G).firstNode(); (v); (v)=(v)->succ())
121 if(v->indeg() == 0)
122 m_root = v;
123 break;
124
125 case rootIsSink:
126 forall_nodes(v,G)for((v)=(G).firstNode(); (v); (v)=(v)->succ())
127 if(v->outdeg() == 0)
128 m_root = v;
129 break;
130
131 case rootIsCenter:
132 {
133 NodeArray<int> degree(G);
134 Queue<node> leaves;
135
136 forall_nodes(v,G)for((v)=(G).firstNode(); (v); (v)=(v)->succ()) {
137 if((degree[v] = v->degree()) == 1)
138 leaves.append(v);
139 }
140
141 while(!leaves.empty()) {
142 v = leaves.pop();
143
144 adjEntry adj;
145 forall_adj(adj, v)for((adj)=(v)->firstAdj(); (adj); (adj)=(adj)->succ()) {
146 node u = adj->twinNode();
147 if(--degree[u] == 1)
148 leaves.append(u);
149 }
150 }
151
152 m_root = v;
153 }
154 break;
155 }
156}

159void RadialTreeLayout::ComputeLevels(const Graph &G)
160{
161 m_parent.init(G);
162 m_level.init(G);
163 m_leaves.init(G,0);
164
165 Queue<node> Q;
166 Stack<node> S;
167
168 Q.append(m_root);
169 m_parent[m_root] = 0;
170 m_level [m_root] = 0;
171
172 int maxLevel = 0;
173
174 while(!Q.empty())
175 {
176 node v = Q.pop();
177 S.push(v);
178 int levelV = m_level[v];
179
180 bool isLeaf = true;
181
182 adjEntry adj;
183 forall_adj(adj, v)for((adj)=(v)->firstAdj(); (adj); (adj)=(adj)->succ()) {
184 node u = adj->twinNode();
185 if(u == m_parent[v])
186 continue;
187
188 isLeaf = false;
189
190 Q.append(u);
191 m_parent[u] = v;
192 m_level[u] = maxLevel = levelV+1;
193 }
194
195 // number of leaves in a subtree rooted at a leaf is 1
196 if(isLeaf)
197 m_leaves[v] = 1.0 / levelV;
198 }
199
200 m_numLevels = maxLevel + 1;
201
202 // compute number of leaves in subtree (already computed for leaves)
203 while(!S.empty())
204 {
205 node v = S.pop();
206 node p = m_parent[v];
207
208 if(p != 0)
209 m_leaves[p] += m_leaves[v];
210 }

468void RadialTreeLayout::Grouping::computeAdd(double &D, double &W)
469{
470 D = W = 0;
471
472 ListIterator<Group> it;
473 for(it = begin(); it.valid(); ++it)
474 {
475 Group &g = *it;
476
477 D += g.m_sumD;
478
479 if(g.m_leafGroup == true)
480 continue;
481
482 W += g.m_sumW;
483
484 ListIterator<Group> itL;
485
486 itL = it.pred();
487 if(itL.valid() == false) {
488 g.m_leftAdd = 0.0;
489 } else {
490 ListIterator<Group> itR = itL.pred();
491 if(itR.valid() == false)
492 g.m_leftAdd = (*itL).m_sumD;
493 else
494 g.m_leftAdd = (*itL).m_sumD * g.m_sumW / (*itR).m_sumW;
495 }
496
497 itL = it.succ();
498 if(itL.valid() == false) {
499 g.m_leftAdd = 0.0;
500 } else {
501 ListIterator<Group> itR = itL.succ();
502 if(itR.valid() == false)
503 g.m_leftAdd = (*itL).m_sumD;
504 else
505 g.m_leftAdd = (*itL).m_sumD * g.m_sumW / (*itR).m_sumW;
506 }
507 }

510// compute grouping for sons of nodes on level i
511void RadialTreeLayout::ComputeGrouping(int i)
512{
513 SListConstIterator<node> it;
514 for(it = m_nodes[i].begin(); it.valid(); ++it)
515 {
516 node v = *it;
517 node p = m_parent[v];
518
519 Grouping &grouping = m_grouping[v];
520 ListIterator<Group> currentGroup;
521
522 adjEntry adj = v->firstAdj();
523 adjEntry adjStop;
524 if(p != 0) {
525 while(adj->twinNode() != p)
526 adj = adj->cyclicSucc();
527 adjStop = adj;
528 adj = adj->cyclicSucc();
529 } else {
530 adjStop = adj;
531 }
532
533 do
534 {
535 node u = adj->twinNode();
536
537 if(!currentGroup.valid() || (*currentGroup).isSameType(u) == false)
538 {
539 currentGroup = grouping.pushBack(Group(this,u));
540
541 } else {
542 (*currentGroup).append(u);
543 }
544
545 adj = adj->cyclicSucc();
546 } while(adj != adjStop);
547 }
