148 * Checks the type of the input given in string
149 * and returns an OgmlAttributeValueId defined in Ogml.h

151 Ogml::AttributeValueId getTypeOfString(const String& input) const
152 {
153 // |--------------------|
154 // | char | ascii-value |
155 // |--------------------|
156 // | '.' | 46 |
157 // | '-' | 45 |
158 // | '+' | 43 |
159 // | '#' | 35 |
160
161 // bool values
162 bool isInt = true;
163 bool isNum = true;
164 bool isHex = true;
165
166 // value for point seperator
167 bool numPoint = false;
168
169 // input is a boolean value
170 if (input == "true" || input == "false" /*|| input == "0" || input == "1"*/)
171 return Ogml::av_bool;
172
173 if (input.length() > 0){
174 char actChar = input[0];
175 int actCharInt = static_cast<int>(actChar);
176 //check the first char
177 if (!isalnum(actChar)){
178
179 if (actCharInt == 35){
180 // support hex values with starting "#"
181 isInt = false;
182 isNum = false;
183 }
184 else
185 {
186
187 // (actChar != '-') and (actChar != '+')
188 if (!(actCharInt == 45) && !(actChar == 43)){
189 isInt = isNum = false;
190 }
191 else
192 {
193 // input[0] == '-' or '+'
194 if (input.length() > 1){
195 // 2nd char have to be a digit or xdigit
196 actChar = input[1];
197 actCharInt = static_cast<int>(actChar);
198 if (!isdigit(actChar)){
199 isInt = false;
200 isNum = false;
201 if (!isxdigit(actChar))
202 return Ogml::av_string;
203 }
204 }
205 else
206 return Ogml::av_string;
207 } // else... (input[0] == '-')
208 }
209 }
210 else{
211 if (!isdigit(actChar)){
212 isInt = false;
213 isNum = false;
214 }
215 if (!isxdigit(actChar)){
216 isHex = false;
217 }
218 }
219
220 // check every input char
221 // and set bool value to false if char-type is wrong
222 for(size_t it=1; ( (it<input.length()) && ((isInt) || (isNum) || (isHex)) ); it++)
223 {
224 actChar = input[it];
225 actCharInt = static_cast<int>(actChar);

229 isInt = false;
230 isHex = false;

236 }// if (actChar == '.')

712 */
713 int validTag(const XmlTagObject &o,
714 Hashing<String, const XmlTagObject*>& ids) const
715 {
716 int valid = Ogml::vs_unexpTag;
717
718 if( o.getName() == getName() ) {
719
720 ListConstIterator<OgmlAttribute*> it;
721 XmlAttributeObject* att;
722
723 //Tag requires attributes
724 if(!compulsiveAttributes.empty()) {
725
726 for(it = compulsiveAttributes.begin(); it.valid(); it++) {
727 //Att not found or invalid
728 if(!o.findXmlAttributeObjectByName((**it).getName(), att) )
729 return valid = Ogml::vs_expAttNotFound;
730 if( (valid = (**it).validAttribute(*att, &o, ids) ) <0 )
731 return valid;
732 //Att is valid
733 att->setValid();
734 }
735 }
736
737 //Choice attributes
738 if(!choiceAttributes.empty()) {
739
740 bool tookChoice = false;
741
742 for(it = choiceAttributes.begin(); it.valid(); it++) {
743 //Choice att found
744 if( o.findXmlAttributeObjectByName((**it).getName(), att) ) {
745 //Proof if valid
746 if( (valid = (**it).validAttribute(*att, &o, ids)) <0 )
747 return valid;
748 tookChoice = true;
749 att->setValid();
750 }
751 }
752
753 if(!tookChoice)
754 return valid = Ogml::vs_expAttNotFound;
755
756 }
757
758 if(!optionalAttributes.empty() && !o.isAttributeLess()) {
759
760 //Check optional attributes
761 for(it = optionalAttributes.begin(); it.valid(); it++) {
762 if( o.findXmlAttributeObjectByName((**it).getName(), att) ) {
763 if( (valid = (**it).validAttribute(*att, &o, ids)) <0 )
764 return valid;
765 att->setValid();
766 }
767 }
768 }
769
770 //Are there still invalid attributes?
771 att = o.m_pFirstAttribute;

1500//
1501// v a l i d a t e
1502//
1503// ********************************************************
1504int OgmlParser::validate(const XmlTagObject * xmlTag, int ogmlTagId)
1505{
1506
1507 OgmlTag *ogmlTag = &s_tags->lookup(ogmlTagId)->info();
1508 ListConstIterator < OgmlTag * > it;
1509 XmlTagObject *sonTag;
1510 int valid;
1511
1512 // Perhaps xmlTag is already valid
1513 if(xmlTag->valid())
1514 return valid = Ogml::vs_valid;
1515
1516 if(!ogmlTag) {
1517 cerr << "Didn't found tag with id \"" << ogmlTagId << "\" in hashtable in OgmlParser::validate! Aborting.\n";
1518 return false;
1519 }
1520
1521 if((valid = ogmlTag->validTag(*xmlTag, m_ids)) < 0) {
1522 this->printValidityInfo(*ogmlTag, *xmlTag, valid, __LINE__1522);
1523 return valid;
1524 }
1525
1526 // if tag ignores its content simply return
1527 if(ogmlTag->ignoresContent()) {
1528 xmlTag->setValid();
1529#ifdef OGDF_DEBUG
1530 this->printValidityInfo(*ogmlTag, *xmlTag, valid = Ogml::vs_valid, __LINE__1530);
1531#endif
1532 return valid = Ogml::vs_valid;
1533 }
1534
1535 // Check if all required son tags exist
1536 if(ogmlTag->ownsCompulsiveTags())
1537 {
1538 // find all obligatoric sons: all obligatoric sons
1539 for (it = ogmlTag->getCompulsiveTags().begin(); it.valid(); it++)
1540 {
1541 int cnt = 0;
1542
1543 // search for untested sons
1544 sonTag = xmlTag->m_pFirstSon;
1545 while(sonTag) {
1546 if(sonTag->getName() == (**it).getName()) {
1547 cnt++;
1548 if((valid = validate(sonTag, (**it).getId())) < 0)
1549 return valid;
1550 }
1551 sonTag = sonTag->m_pBrother;
1552 }
1553
1554 // Exp. son not found
1555 if(cnt == 0) {
1556 this->printValidityInfo(*ogmlTag, *xmlTag, valid = Ogml::vs_expTagNotFound, __LINE__1556);
1557 return valid;
1558 }
1559
1560 // Check cardinality
1561 if(cnt < (**it).getMinOccurs() || cnt > (**it).getMaxOccurs()) {
1562 this->printValidityInfo((**it), *xmlTag, valid = Ogml::vs_cardErr, __LINE__1562);
1563 return valid;
1564 }
1565 }
1566 }
1567
1568 // Check if choice son tags exist
1569 if(ogmlTag->ownsChoiceTags())
1570 {
1571 bool tookChoice = false;
1572
1573 // find all obligatoric sons: all obligatoric sons
1574 for (it = ogmlTag->getChoiceTags().begin(); it.valid(); it++)
1575 {
1576 int cnt = 0;
1577
1578 // search for untested sons
1579 sonTag = xmlTag->m_pFirstSon;
1580 while(sonTag) {
1581 if(sonTag->getName() == (**it).getName()) {
1582 if((valid = validate(sonTag, (**it).getId())) < 0)
1583 return valid;
1584 tookChoice = true;
1585 cnt++;
1586 }
1587 sonTag = sonTag->m_pBrother;
1588 }
1589
1590 // Check cardinality
1591 if(cnt > 0 && (cnt < (**it).getMinOccurs()
1592 || cnt > (**it).getMaxOccurs())) {
1593 this->printValidityInfo((**it), *xmlTag, valid = Ogml::vs_cardErr, __LINE__1593);
1594 return valid;
1595 }
1596
1597 }
1598 if ((!tookChoice) && (xmlTag->m_pFirstSon)) {
1599 this->printValidityInfo((**it), *xmlTag, valid = Ogml::vs_tagEmptIncl, __LINE__1599);
1600 return valid;
1601 }
1602
1603 } //Check choice son tags
1604
1605 // Check if opt son tags exist
1606 if(ogmlTag->ownsOptionalTags())
1607 {
1608 // find all obligatoric sons: all obligatoric sons
1609 for (it = ogmlTag->getOptionalTags().begin(); it.valid(); ++it)
1610 {
1611 int cnt = 0;
1612
1613 // search for untested sons
1614 sonTag = xmlTag->m_pFirstSon;
1615 while(sonTag) {
1616
1617 if(sonTag->getName() == (**it).getName()) {
1618 if((valid = validate(sonTag, (**it).getId())) < 0)
1619 return valid;
1620 cnt++;
1621 }
1622 sonTag = sonTag->m_pBrother;
1623 }
1624
1625 // Check cardinality
1626 // if( (cnt<(**it).getMinOccurs() || cnt>(**it).getMaxOccurs()) ) {
1627 if(cnt > (**it).getMaxOccurs()) {
1628 this->printValidityInfo((**it), *xmlTag, valid = Ogml::vs_cardErr, __LINE__1628);
1629 return valid;
1630 }
1631 }
1632 }
1633
1634 // Are there invalid son tags left?
1635 sonTag = xmlTag->m_pFirstSon;
1636 while(sonTag) {
1637 // tag already valid
1638 if(!sonTag->valid()) {
1639 this->printValidityInfo(*ogmlTag, *xmlTag, valid = Ogml::vs_unexpTag, __LINE__1639);
1640 return valid;
1641 }
1642 sonTag = sonTag->m_pBrother;
1643 }
1644
1645 // Finally xmlTag is valid :-)
1646 xmlTag->setValid();
1647
1648#ifdef OGDF_DEBUG
1649 this->printValidityInfo(*ogmlTag, *xmlTag, valid = Ogml::vs_valid, __LINE__1649);
1650#endif
1651
1652 return Ogml::vs_valid;
1653}

1828// ***********************************************************
1829//
1830// c h e c k G r a p h T y p e
1831//
1832// ***********************************************************
1833bool OgmlParser::checkGraphType(const XmlTagObject *xmlTag) const
1834{
1835 if(xmlTag->getName() != Ogml::s_tagNames[Ogml::t_ogml]) {
1836 cerr << "ERROR: Expecting root tag \"" << Ogml::s_tagNames[Ogml::t_ogml] << "\" in OgmlParser::checkGraphType!\n";
1837 return false;
1838 }
1839
1840 // Normal graph present
1841 if(!isGraphHierarchical(xmlTag)) {
1842 m_graphType = Ogml::graph;
1843 return true;
1844 }
1845
1846 // Cluster-/Compound graph present
1847 m_graphType = Ogml::clusterGraph;
1848
1849 // Traverse the parse tree and collect all edge tags
1850 List<const XmlTagObject*> edges;
1851 if(xmlTag->getName() == Ogml::s_tagNames[Ogml::t_edge]) edges.pushBack(xmlTag);
1852 XmlTagObject* son = xmlTag->m_pFirstSon;
1853 while(son) {
1854 if(son->getName() == Ogml::s_tagNames[Ogml::t_edge]) edges.pushBack(son);
1855 son = son->m_pBrother;
1856 }
1857
1858 // Cluster graph already present
1859 if(edges.empty()) return true;
1860
1861 // Traverse edges
1862 ListConstIterator<const XmlTagObject*> edgeIt;
1863 for(edgeIt = edges.begin(); edgeIt.valid() && m_graphType != Ogml::compoundGraph; edgeIt++)
1864 {
1865 // Traverse the sources/targets
1866 son = (*edgeIt)->m_pFirstSon;
1867
1868 // Parse tree is valid so one edge contains at least one source/target
1869 // with idRef attribute
1870 while(son) {
1871 XmlAttributeObject* att;
1872 if(son->findXmlAttributeObjectByName(Ogml::s_attributeNames[Ogml::a_nodeIdRef], att)) {
1873 const XmlTagObject *refTag = m_ids.lookup(att->getValue())->info();
1874 if(isNodeHierarchical(refTag)) {
1875 m_graphType = Ogml::compoundGraph;
1876 break;
1877 }
1878 }
1879 son = son->m_pBrother;
1880 }
1881 }
1882
1883 return true;
1884}

2111{
2112 String output;

2116 if (str[i] == '&')
2117 {
2118 if (i+3 < str.length())
2119 {
2120 if ((str[i+1] == 'l') && (str[i+2] == 't') && (str[i+3] == ';')){
2121 // found char sequence "&lt;"
2122 output += "<";
2123 } else {
2124 if ((str[i+1] == 'g') && (str[i+2] == 't') && (str[i+3] == ';')){
2125 // found char sequence "&gt;"
2126 // \n newline is required!!!
2127 output += ">\n";
2128 }
2129 }
2130 i = i + 4;
2131 }
2132 } else {
2133 char c = str[i];
2134 output += c;
2135 i++;
2136 }
2137 }
2138 str += "\n";
2139 return output;
2140}
