111void System::init()
112{
113 s_cpuFeatures = 0;
114 s_cacheSize = 0;
115 s_cacheLine = 0;
116
117 // currently not working for shared libs on Linux
118#if !defined(OGDF_DLL) || !defined(OGDF_SYSTEM_UNIX)
119
120 int CPUInfo[4] = {-1};
121 __cpuid(CPUInfo, 0);
122
123 unsigned int nIds = CPUInfo[0];
124 if(nIds >= 1)
125 {
126 __cpuid(CPUInfo, 1);
127
128 int featureInfoECX = CPUInfo[2];
129 int featureInfoEDX = CPUInfo[3];
130
131 if(featureInfoEDX & (1 << 23)) s_cpuFeatures |= cpufmMMX;
132 if(featureInfoEDX & (1 << 25)) s_cpuFeatures |= cpufmSSE;
133 if(featureInfoEDX & (1 << 26)) s_cpuFeatures |= cpufmSSE2;
134 if(featureInfoECX & (1 << 0)) s_cpuFeatures |= cpufmSSE3;
135 if(featureInfoECX & (1 << 9)) s_cpuFeatures |= cpufmSSSE3;
136 if(featureInfoECX & (1 << 19)) s_cpuFeatures |= cpufmSSE4_1;
137 if(featureInfoECX & (1 << 20)) s_cpuFeatures |= cpufmSSE4_2;
138 if(featureInfoECX & (1 << 5)) s_cpuFeatures |= cpufmVMX;
139 if(featureInfoECX & (1 << 6)) s_cpuFeatures |= cpufmSMX;
140 if(featureInfoECX & (1 << 7)) s_cpuFeatures |= cpufmEST;
141 if(featureInfoECX & (1 << 3)) s_cpuFeatures |= cpufmMONITOR;
142 }
143
144 __cpuid(CPUInfo, 0x80000000);
145 unsigned int nExIds = CPUInfo[0];
146
147 if(nExIds >= 0x80000006) {
148 __cpuid(CPUInfo, 0x80000006);
149 s_cacheLine = CPUInfo[2] & 0xff;
150 s_cacheSize = (CPUInfo[2] >> 16) & 0xffff;
151 }
152
153#if defined(OGDF_SYSTEM_WINDOWS) || defined(__CYGWIN__)
154 QueryPerformanceFrequency(&s_HPCounterFrequency);
155
156 SYSTEM_INFO siSysInfo;
157 GetSystemInfo(&siSysInfo);
158 s_pageSize = siSysInfo.dwPageSize;
159 s_numberOfProcessors = siSysInfo.dwNumberOfProcessors;
160
161#elif defined(OGDF_SYSTEM_UNIX) && defined(__APPLE__1)
162 unsigned long long value;
163 size_t size = sizeof( value );
164 if (sysctlbyname("hw.pagesize", &value, &size, NULL__null, 0) !=-1)
165 s_pageSize = (int)value;
166 else
167 s_pageSize = 0;
168
169 if (sysctlbyname("hw.ncpu", &value, &size, NULL__null, 0) !=-1)
170 s_numberOfProcessors = (int)value;
171 else
172 s_numberOfProcessors = 1;
173
174#elif defined(OGDF_SYSTEM_UNIX)
175 s_pageSize = sysconf(_SC_PAGESIZE29);
176 s_numberOfProcessors = (int)sysconf(_SC_NPROCESSORS_CONF57);
177
178#else
179 s_pageSize = 0; // just a placeholder!!!
180 s_numberOfProcessors = 1; // just a placeholder!!!
181#endif
182
183#endif
184}

358size_t System::memoryAllocatedByMalloc()
359{
360 _HEAPINFO hinfo;
361 int heapstatus;
362 hinfo._pentry = NULL__null;
363
364 size_t allocMem = 0;
365 while((heapstatus = _heapwalk(&hinfo)) == _HEAPOK)
366 {
367 if(hinfo._useflag == _USEDENTRY)
368 allocMem += hinfo._size;
369 }
370
371 return allocMem;
372}
