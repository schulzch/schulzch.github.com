677 adj1 = adj1->cyclicSucc();
678
679 // if the actual node has an edge to the deleted node
680 // increase the visited value for the actual node...
681 if (adj1->twinNode() == vk){
682 cto.incVisited(u);
683 // ... and skip the actual adjEntry
684 adj1 = adj1->cyclicSucc();
685 }
686 adjSucc[u] = adj1;
687
688 // add actual node to outerNodes[f]
689 for (adj2 = adjPred[u]; adj2 != adjSucc[u]; adj2 = adj2->cyclicPred()){
690 cto.addOuterNode(u, E.leftFace(adj2));
691 }
692 adj1 = adj1->twin();
693 }
694
695 if (!cto.isNode()){
696 if ( ((adjSucc[cl])->twinNode() == cr)
697 && ( cto.isOnlyEdge(E.rightFace(adjSucc[cl])) ) ){
698 cto.decSepf(cl);
699 cto.decSepf(cr);
700 }
701 }
702
703 // update cto
704 cto.doUpdate();
705
706 #ifdef OUTPUT_TSO
707 cto.output();
708 #endif
709 }// while (cto.isPossible())
710
711 // finally push the base (v1,v2) to the order
712 V = ShellingOrderSet(2);
713 V[1] = v1;
714 V[2] = v2;
715 partition.pushFront(V);
