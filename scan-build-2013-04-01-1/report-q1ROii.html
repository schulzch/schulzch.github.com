1668int MultiEdgeApproxInserter::findShortestPath(node s, node t)
1669{
1670 NodeArray<adjEntry> spPred(m_dual, 0);
1671 QueuePure<adjEntry> queue;
1672 int oldIdCount = m_dual.maxEdgeIndex();
1673
1674 // augment dual by edges from s to all adjacent faces of s ...
1675 adjEntry adj;
1676 forall_adj(adj,s)for((adj)=(s)->firstAdj(); (adj); (adj)=(adj)->succ()) {
1677 // starting edges of bfs-search are all edges leaving s
1678 edge eDual = m_dual.newEdge(m_vS, m_faceNode[m_E.rightFace(adj)]);
1679 m_primalAdj[eDual->adjSource()] = adj;
1680 m_primalAdj[eDual->adjTarget()] = 0;
1681 queue.append(eDual->adjSource());
1682 }
1683
1684 // ... and from all adjacent faces of t to t
1685 forall_adj(adj,t)for((adj)=(t)->firstAdj(); (adj); (adj)=(adj)->succ()) {
1686 edge eDual = m_dual.newEdge(m_faceNode[m_E.rightFace(adj)], m_vT);
1687 m_primalAdj[eDual->adjSource()] = adj;
1688 m_primalAdj[eDual->adjTarget()] = 0;
1689 }
1690
1691 // actual search (using bfs on directed dual)
1692 int len = -2;
1693 for( ; ;)
1694 {
1695 // next candidate edge
1696 adjEntry adjCand = queue.pop();
1697 node v = adjCand->twinNode();
1698
1699 // leads to an unvisited node?
1700 if (spPred[v] == 0)
1701 {
1702 // yes, then we set v's predecessor in search tree
1703 spPred[v] = adjCand;
1704
1705 // have we reached t ...
1706 if (v == m_vT)
1707 {
1708 // ... then search is done.
1709
1710 adjEntry adj;
1711 do {
1712 adj = spPred[v];
1713 ++len;
1714 v = adj->theNode();
1715 } while(v != m_vS);
1716
1717 break;
1718 }
1719
1720 // append next candidate edges to queue
1721 // (all edges leaving v)
1722 adjEntry adj;
1723 forall_adj(adj,v)for((adj)=(v)->firstAdj(); (adj); (adj)=(adj)->succ()) {
1724 if(adj->twinNode() != m_vS)
1725 queue.append(adj);
1726 }
1727 }
1728 }
1729
1730 // remove augmented edges again
1731 while ((adj = m_vS->firstAdj()) != 0)
1732 m_dual.delEdge(adj->theEdge());
1733
1734 while ((adj = m_vT->firstAdj()) != 0)
1735 m_dual.delEdge(adj->theEdge());
1736
1737 m_dual.resetEdgeIdCount(oldIdCount);
1738
1739 return len;
1740}
