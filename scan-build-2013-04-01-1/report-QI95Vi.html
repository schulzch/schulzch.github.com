416edge VariableEmbeddingInserter::crossedEdge(adjEntry adj) const
417{
418 edge e = adj->theEdge();
419
420 adj = adj->cyclicSucc();
421 while(adj->theEdge() == e)

428int VariableEmbeddingInserter::costCrossed(edge eOrig) const
429{
430 int c = 0;
431
432 const List<edge> &L = m_pPG->chain(eOrig);
433
434 ListConstIterator<edge> it = L.begin();
435 if(m_edgeSubgraph != 0) {
436 for(++it; it.valid(); ++it) {
437 int counter = 0;
438 edge e = m_pPG->original(crossedEdge((*it)->adjSource()));
439 for(int i=0; i<32; i++)
440 if((*m_edgeSubgraph)[eOrig] & (*m_edgeSubgraph)[e] & (1<<i))
441 counter++;
442 c += counter*(*m_costOrig)[e];
443 }
444 c *= m_bigM;
445 if(c == 0)
446 c = 1;
447 } else {
448 for(++it; it.valid(); ++it) {
449 c += (*m_costOrig)[m_pPG->original(crossedEdge((*it)->adjSource()))];
450 }
451 }
452
453 return c;
454}

457//-------------------------------------------------------------------
458// find optimal edge insertion path from s to t in connected
459// graph G
460//-------------------------------------------------------------------
461
462void VariableEmbeddingInserter::insert(node s,
463 node t,
464 SList<adjEntry> &eip)
465{
466 PlanRep &PG = *m_pPG;
467 eip.clear();
468
469 m_s = s; m_t = t;
470 m_pEip = &eip;
471
472 // compute biconnected components of PG
473 EdgeArray<int> compnum(PG);
474 int c = biconnectedComponents(PG,compnum);
475
476 m_compV.init(PG);
477 m_nodeB.init(c);
478
479 // edgeB[i] = list of edges in component i
480 m_edgeB.init(c);
481 edge e;
482 forall_edges(e,PG)for((e)=(PG).firstEdge(); (e); (e)=(e)->succ())
483 m_edgeB[compnum[e]].pushBack(e);
484
485 // construct arrays compV and nodeB such that
486 // m_compV[v] = list of components containing v
487 // m_nodeB[i] = list of vertices in component i
488 NodeArray<bool> mark(PG,false);
489
490 int i;
491 for(i = 0; i < c; ++i) {
492 SListConstIterator<edge> itEdge;
493 for(itEdge = m_edgeB[i].begin(); itEdge.valid(); ++itEdge)
494 {
495 edge e = *itEdge;
496
497 if (!mark[e->source()]) {
498 mark[e->source()] = true;
499 m_nodeB[i].pushBack(e->source());
500 }
501 if (!mark[e->target()]) {
502 mark[e->target()] = true;
503 m_nodeB[i].pushBack(e->target());
504 }
505 }
506
507 SListConstIterator<node> itNode;
508 for(itNode = m_nodeB[i].begin(); itNode.valid(); ++itNode)
509 {
510 node v = *itNode;
511 m_compV[v].pushBack(i);
512 mark[v] = false;
513 }
514 }
515 mark.init();
516
517 // find path from s to t in BC-tree
518 // call of blockInsert() is done in dfs_vertex() when we have found the
519 // path and we return from the recursion.
520 // if no path is found, s and t are in different connected components
521 // and thus an empty edge insertion path is correct!
522 m_GtoBC.init(PG,0);
523 dfsVertex(s,-1);
524
525 // deallocate resources used by insert()
526 m_GtoBC.init();
527 m_edgeB.init();
528 m_nodeB.init();
529 m_compV.init();
530}

569//-------------------------------------------------------------------
570// recursive path search from s to t in BC-tree (vertex case)
571//-------------------------------------------------------------------
572
573bool VariableEmbeddingInserter::dfsVertex(node v, int parent)
574{
575 // forall biconnected components containing v (except predecessor parent)
576 SListConstIterator<int> itI;
577 for(itI = m_compV[v].begin(); itI.valid(); ++itI)
578 {
579 int i = *itI;
580
581 if (i == parent) continue;
582
583 node repT; // representative of t in B(i)
584 if (dfsComp(i,v,repT) == true) { // path found?
585 // build graph BC of biconnected component B(i)
586 SList<node> nodesG;
587 BiconnectedComponent BC;
588
589 SListConstIterator<edge> itE;
590 for(itE = m_edgeB[i].begin(); itE.valid(); ++itE)
591 {
592 edge e = *itE;
593
594 if (m_GtoBC[e->source()] == 0) {
595 m_GtoBC[e->source()] = BC.newNode();
596 nodesG.pushBack(e->source());
597 }
598 if (m_GtoBC[e->target()] == 0) {
599 m_GtoBC[e->target()] = BC.newNode();
600 nodesG.pushBack(e->target());
601 }
602
603 edge eBC = BC.newEdge(m_GtoBC[e->source()],m_GtoBC[e->target()]);
604 BC.m_BCtoG[eBC->adjSource()] = e->adjSource();
605 BC.m_BCtoG[eBC->adjTarget()] = e->adjTarget();
606
607 BC.typeOf(eBC, m_forbidCrossingGens ? ((PlanRepUML*)m_pPG)->typeOf(e) : Graph::association);
608 edge eOrig = m_pPG->original(e);
609 if(m_costOrig != 0) {
610 if(m_edgeSubgraph != 0) {
611 int counter = 0;
612 for(int i = 0; i<32; i++)
613 if((*m_edgeSubgraph)[m_st] & (*m_edgeSubgraph)[eOrig] & (1<<i))
614 counter++;
615 counter *= m_bigM;
616 int cost = counter * (*m_costOrig)[eOrig];
617 if(cost == 0)
618 cost = 1;
619 BC.cost(eBC, cost);
620 } else
621 BC.cost(eBC, (eOrig == 0) ? 0 : (*m_costOrig)[eOrig]);
622 }
623 }
624
625 // less than 3 nodes requires no crossings (cannot build SPQR-tree
626 // for a graph with less than 3 nodes!)
627 if (nodesG.size() >= 3) {
628 List<adjEntry> L;
629 blockInsert(BC,m_GtoBC[v],m_GtoBC[repT],L); // call biconnected case
630
631 // transform crossed edges to edges in G
632 ListConstIterator<adjEntry> it;
633 for(it = L.rbegin(); it.valid(); --it) {
634 m_pEip->pushFront(BC.m_BCtoG[*it]);
635 }
636 }
637
638 // set entries of GtoBC back to nil (GtoBC allocated only once
639 // in insert()!)
640 SListConstIterator<node> itV;
641 for(itV = nodesG.begin(); itV.valid(); ++itV)
642 m_GtoBC[*itV] = 0;
643
644 return true; // path found
645 }
646 }
647
648 return false; // path not found
649}

653//-------------------------------------------------------------------
654// recursive path search from s to t in BC-tree (component case)
655//-------------------------------------------------------------------
656
657bool VariableEmbeddingInserter::dfsComp(int i, node parent, node &repT)
658{
659 // forall nodes in biconected component B(i) (except predecessor parent)
660 SListConstIterator<node> it;
661 for(it = m_nodeB[i].begin(); it.valid(); ++it)
662 {
663 repT = *it;
664
665 if (repT == parent) continue;
666 if (repT == m_t) { // t found?
667 return true;
668 }
669 if (dfsVertex(repT,i) == true) {
670 return true; // path found
671 }
672 }
673
674 return false; // path not found
675}

737// build expanded graph (by expanding skeleton(v), edges eIn and eOut
738// are the adjacent tree edges on the path from v1 to v2
739//-------------------------------------------------------------------
740
741void ExpandedGraph::expand(node v, edge eIn, edge eOut)
742{
743 m_exp.clear();
744 while (!m_nodesG.empty())
745 m_GtoExp[m_nodesG.popBackRet()] = 0;
746
747 const Skeleton &S = m_T.skeleton(v);
748
749 if (eIn != 0) {
750 edge eInS = (v != eIn->source()) ? m_T.skeletonEdgeTgt(eIn) :
751 m_T.skeletonEdgeSrc(eIn);
752 node x = S.original(eInS->source()), y = S.original(eInS->target());
753 m_eS = insertEdge(x,y,0);
754 }
755 if (eOut != 0) {
756 edge eOutS = (v != eOut->source()) ? m_T.skeletonEdgeTgt(eOut) :
757 m_T.skeletonEdgeSrc(eOut);
758 node x = S.original(eOutS->source()), y = S.original(eOutS->target());
759 m_eT = insertEdge(x,y,0);
760 }
761
762 expandSkeleton(v, eIn, eOut);
763
764 planarEmbed(m_exp);
765 m_E.init(m_exp);
766}

769//-------------------------------------------------------------------
770// expand one skeleton (recursive construction)
771//-------------------------------------------------------------------
772
773void ExpandedGraph::expandSkeleton(node v, edge e1, edge e2)
774{
775 const StaticSkeleton &S = *dynamic_cast<StaticSkeleton*>(&m_T.skeleton(v));
776 const Graph &M = S.getGraph();
777
778 edge e;
779 forall_edges(e,M)for((e)=(M).firstEdge(); (e); (e)=(e)->succ())
780 {
781 edge eG = S.realEdge(e);
782 if (eG != 0) {
783 insertEdge(eG->source(),eG->target(),eG);
784
785 } else {
786 edge eT = S.treeEdge(e);
787
788 // do not expand virtual edges corresponding to tree edges e1 or e2
789 if (eT != e1 && eT != e2) {
790 expandSkeleton((v == eT->source()) ? eT->target() : eT->source(),
791 eT,0);
792 }
793 }
794 }
795}

798//-------------------------------------------------------------------
799// insert edge in exp (from a node corresponding to vG in G to a node
800// corresponding to wG)
801//-------------------------------------------------------------------
802
803edge ExpandedGraph::insertEdge(node vG, node wG, edge eG)
804{
805 node &rVG = m_GtoExp[vG];
806 node &rWG = m_GtoExp[wG];
807
808 if (rVG == 0) {
809 rVG = m_exp.newNode();
810 m_nodesG.pushBack(vG);
811 }
812 if (rWG == 0) {
813 rWG = m_exp.newNode();
814 m_nodesG.pushBack(wG);
815 }
816
817 edge e1 = m_exp.newEdge(rVG,rWG);
818
819 if(eG != 0) {
820 m_expToG[e1->adjSource()] = eG->adjSource();
821 m_expToG[e1->adjTarget()] = eG->adjTarget();
822 } else {
823 m_expToG[e1->adjSource()] = 0;
824 m_expToG[e1->adjTarget()] = 0;
825 }
826
827 return e1;
828}

831//-------------------------------------------------------------------
832// construct augmented dual of exp
833//-------------------------------------------------------------------
834
835void ExpandedGraph::constructDual(node s, node t,
836 GraphCopy &GC, const EdgeArray<bool> *forbiddenEdgeOrig)
837{
838 m_dual.clear();
839
840 FaceArray<node> faceNode(m_E);
841
842 // constructs nodes (for faces in exp)
843 face f;
844 forall_faces(f,m_E)for((f)=(m_E).firstFace(); (f); (f)=(f)->succ()) {
845 faceNode[f] = m_dual.newNode();
846 }
847
848 // construct dual edges (for primal edges in exp)
849 node v;
850 forall_nodes(v,m_exp)for((v)=(m_exp).firstNode(); (v); (v)=(v)->succ())
851 {
852 adjEntry adj;
853 forall_adj(adj,v)for((adj)=(v)->firstAdj(); (adj); (adj)=(adj)->succ())
854 {
855 // cannot cross edges that does not correspond to real edges
856 adjEntry adjG = m_expToG[adj];
857 if(adjG == 0)
858 continue;
859
860 // Do not insert edges into dual if crossing the original edge
861 // is forbidden
862 if(forbiddenEdgeOrig &&
863 (*forbiddenEdgeOrig)[GC.original(m_BC.m_BCtoG[m_expToG[adj]]->theEdge())] == true)
864 continue;
865
866 node vLeft = faceNode[m_E.leftFace (adj)];
867 node vRight = faceNode[m_E.rightFace(adj)];
868
869 m_primalEdge[m_dual.newEdge(vLeft,vRight)] = adj;
870 }
871 }
872
873 // augment dual by m_vS and m_vT
874 m_vS = m_dual.newNode();
875 if (m_GtoExp[s] != 0)
876 {
877 adjEntry adj;
878 forall_adj(adj,m_GtoExp[s])for((adj)=(m_GtoExp[s])->firstAdj(); (adj); (adj)=(adj)->
succ())
879 m_dual.newEdge(m_vS,faceNode[m_E.rightFace(adj)]);
880 }
881 else
882 {
883 m_dual.newEdge(m_vS,faceNode[m_E.rightFace(m_eS->adjSource())]);
884 m_dual.newEdge(m_vS,faceNode[m_E.rightFace(m_eS->adjTarget())]);
885 }
886
887 m_vT = m_dual.newNode();
888 if (m_GtoExp[t] != 0)
889 {
890 adjEntry adj;
891 forall_adj(adj,m_GtoExp[t])for((adj)=(m_GtoExp[t])->firstAdj(); (adj); (adj)=(adj)->
succ())
892 m_dual.newEdge(faceNode[m_E.rightFace(adj)], m_vT);
893 }
894 else
895 {
896 m_dual.newEdge(faceNode[m_E.rightFace(m_eT->adjSource())], m_vT);
897 m_dual.newEdge(faceNode[m_E.rightFace(m_eT->adjTarget())], m_vT);
898 }
899}

980//-------------------------------------------------------------------
981// find shortest path in dual from m_vS to m_vT; output this path
982// in L by omitting first and last edge, and translating edges to G
983//-------------------------------------------------------------------
984
985void ExpandedGraph::findShortestPath(Graph::EdgeType eType, List<adjEntry> &L)
986{
987 NodeArray<edge> spPred(m_dual,0); // predecessor in shortest path tree
988 List<edge> queue; // candidate edges
989
990 // start with all edges leaving from m_vS
991 edge e;
992 forall_adj_edges(e,m_vS)for(ogdf::adjEntry ogdf_loop_var=(m_vS)->firstAdj(); ogdf::
test_forall_adj_edges(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var
->succ())
993 queue.pushBack(e);
994
995 for( ; ; ) {
996 edge eCand = queue.popFrontRet(); // next candidate from front of queue
997 node v = eCand->target();
998
999 // hit an unvisited node ?
1000 if (spPred[v] == 0) {
1001 spPred[v] = eCand;
1002
1003 // if it is m_vT, we have found the shortest path
1004 if (v == m_vT) {
1005 // build path from shortest path tree
1006 while(v != m_vS) {
1007 adjEntry adjExp = m_primalEdge[spPred[v]];
1008 if (adjExp != 0) // == nil for first and last edge
1009 L.pushFront(m_expToG[adjExp]);
1010 v = spPred[v]->source();
1011 }
1012 return;
1013 }
1014
1015 // append next candidates to end of queue
1016 forall_adj_edges(e,v)for(ogdf::adjEntry ogdf_loop_var=(v)->firstAdj(); ogdf::test_forall_adj_edges
(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var->succ()) {
1017 if(v == e->source() &&
1018 (eType != Graph::generalization || m_primalIsGen[e] == false))
1019 {
1020 queue.pushBack(e);
1021 }
1022 }
1023 }
1024 }
1025}

1028//-------------------------------------------------------------------
1029// find weighted shortest path in dual from m_vS to m_vT; output this path
1030// in L by omitting first and last edge, and translating edges to G
1031//-------------------------------------------------------------------
1032
1033void ExpandedGraph::findWeightedShortestPath(Graph::EdgeType eType,
1034 List<adjEntry> &L)
1035{
1036 int maxCost = 0;
1037 edge eDual;
1038 forall_edges(eDual,m_dual)for((eDual)=(m_dual).firstEdge(); (eDual); (eDual)=(eDual)->
succ()) {
1039 int c = costDual(eDual);
1040 if (c > maxCost) maxCost = c;
1041 }
1042
1043 ++maxCost;
1044 Array<SListPure<edge> > nodesAtDist(maxCost);
1045
1046 NodeArray<edge> spPred(m_dual,0); // predecessor in shortest path tree
1047 //List<edge> queue; // candidate edges
1048
1049 // start with all edges leaving from m_vS
1050 edge e;
1051 forall_adj_edges(e,m_vS)for(ogdf::adjEntry ogdf_loop_var=(m_vS)->firstAdj(); ogdf::
test_forall_adj_edges(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var
->succ())
1052 nodesAtDist[0].pushBack(e);
1053
1054 // actual search (using extended bfs on directed dual)
1055 int currentDist = 0;
1056 for( ; ; ) {
1057 // next candidate edge
1058 while(nodesAtDist[currentDist % maxCost].empty())
1059 ++currentDist;
1060
1061 edge eCand = nodesAtDist[currentDist % maxCost].popFrontRet();
1062 node v = eCand->target();
1063
1064 // leads to an unvisited node ?
1065 if (spPred[v] == 0) {
1066 // yes, then we set v's predecessor in search tree
1067 spPred[v] = eCand;
1068
1069 // have we reached t ...
1070 if (v == m_vT) {
1071 // ... then search is done.
1072 // construct list of used edges (translated to crossed
1073 // adjacency entries in G)
1074 while(v != m_vS) {
1075 adjEntry adjExp = m_primalEdge[spPred[v]];
1076 if (adjExp != 0) // == nil for first and last edge
1077 L.pushFront(m_expToG[adjExp]);
1078 v = spPred[v]->source();
1079 }
1080 return;
1081 }
1082
1083 // append next candidates to end of queue
1084 // (all edges leaving v)
1085 forall_adj_edges(e,v)for(ogdf::adjEntry ogdf_loop_var=(v)->firstAdj(); ogdf::test_forall_adj_edges
(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var->succ()) {
1086 if(v == e->source() &&
1087 (eType != Graph::generalization || m_primalIsGen[e] == false))
1088 {
1089 int listPos = (currentDist + costDual(e)) % maxCost;
1090 nodesAtDist[listPos].pushBack(e);
1091 }
1092 }
1093 }
1094 }
1095}

1098//-------------------------------------------------------------------
1099// find optimal edge insertion path from s to t for biconnected
1100// graph G (OptimalBlockInserter)
1101//-------------------------------------------------------------------
1102
1103void VariableEmbeddingInserter::blockInsert(const BiconnectedComponent &BC,
1104 node s,
1105 node t,
1106 List<adjEntry> &L)
1107{
1108 L.clear();
1109
1110 // construct SPQR-tree
1111 StaticPlanarSPQRTree T(BC);
1112 const Graph &tree = T.tree();
1113
1114
1115 // find allocation nodes of s and t and representatives in skeletons
1116 NodeArray<node> containsS(tree,0);
1117 NodeArray<node> containsT(tree,0);
1118
1119 node v, w;
1120 forall_nodes(v,tree)for((v)=(tree).firstNode(); (v); (v)=(v)->succ()) {
1121 const Skeleton &S = T.skeleton(v);
1122 const Graph &M = S.getGraph();
1123
1124 forall_nodes(w,M)for((w)=(M).firstNode(); (w); (w)=(w)->succ()) {
1125 if (S.original(w) == s)
1126 containsS[m_v1 = v] = w;
1127 if (S.original(w) == t)
1128 containsT[m_v2 = v] = w;
1129 }
1130 }
1131
1132 // find path in tree from an allocation node m_v1 of s to an
1133 // allocation m_v2 of t
1134 List<edge> path;
1135 pathSearch(m_v1,0,path);
1136
1137 // remove unnecessary allocation nodes of s from start of path
1138 while(!path.empty() && containsS[w = path.front()->opposite(m_v1)] != 0)
1139 {
1140 m_v1 = w;
1141 path.popFront();
1142 }
1143
1144 // remove unnecessary allocation nodes of t from end of path
1145 while(!path.empty() && containsT[w = path.back()->opposite(m_v2)] != 0)
1146 {
1147 m_v2 = w;
1148 path.popBack();
1149 }
1150
1151 // call build_subpath for every R-node building the list L of crossed edges
1152 ExpandedGraph Exp(BC,T);
1153
1154 if (T.typeOf(m_v1) == SPQRTree::RNode)
1155 buildSubpath(m_v1, 0, (path.empty()) ? 0 : path.front(), L, Exp, s, t);
1156
1157 v = m_v1;
1158 ListConstIterator<edge> it;
1159 for(it = path.begin(); it.valid(); ++it)
1160 {
1161 edge e = *it;
1162 v = e->opposite(v);
1163
1164 if (T.typeOf(v) == SPQRTree::RNode)
1165 buildSubpath(v, e,
1166 (it.succ().valid() == false) ? 0 : *(it.succ()), L, Exp, s, t);
1167 }
1168}

1171//-------------------------------------------------------------------
1172// recursive search for path from v1 to v2 in tree
1173//-------------------------------------------------------------------
1174
1175bool VariableEmbeddingInserter::pathSearch(node v, edge parent, List<edge> &path)
1176{
1177 if (v == m_v2)
1178 return true;
1179
1180 edge e;
1181 forall_adj_edges(e,v)for(ogdf::adjEntry ogdf_loop_var=(v)->firstAdj(); ogdf::test_forall_adj_edges
(ogdf_loop_var,(e)); ogdf_loop_var=ogdf_loop_var->succ()) {
1182 if (e == parent) continue;
1183 if (pathSearch(e->opposite(v),e,path) == true) {
1184 path.pushFront(e);
1185 return true;
1186 }
1187 }
1188
1189 return false;
1190}

1193//-------------------------------------------------------------------
1194// find the shortest path from represent. of s to represent. of t in
1195// the dual of the (partially) expanded skeleton of v
1196//-------------------------------------------------------------------
1197
1198void VariableEmbeddingInserter::buildSubpath(
1199 node v,
1200 edge eIn,
1201 edge eOut,
1202 List<adjEntry> &L,
1203 ExpandedGraph &Exp,
1204 node s,
1205 node t)
1206{
1207 // build expanded graph Exp
1208 Exp.expand(v,eIn,eOut);
1209
1210 // construct augmented dual of expanded graph
1211 if(m_forbidCrossingGens)
1212 Exp.constructDualForbidCrossingGens(s,t);
1213 else

1224}
