682bool XmlParser::read(Graph &G)
683{
684 G.clear();
685
686 int minId, maxId, nodetypeCount;
687 XmlObject *graphObject = getNodeIdRange(minId, maxId, nodetypeCount,0);
688
689 //cout << endl << minId << " " << maxId << endl;
690 if (!graphObject) {
691 setError("missing graph key");
692 return false;
693 }
694
695 Array<double> typeWidth(0,nodetypeCount,0);
696 Array<double> typeHeight(0,nodetypeCount,0);
697 Array<char*> typeName(nodetypeCount+1);
698 Array<char*> idMap(maxId+1);
699 if (!makeIdMap(maxId,idMap,nodetypeCount,typeName,typeWidth,typeHeight,graphObject)) {
700 setError("wrong name identifier");
701 return false;
702 }
703
704 Array<node> mapToNode(minId,maxId,0);
705 int notDefined = minId-1; //indicates not defined id key
706 int idCount = minId;
707
708 XmlObject *son = graphObject->m_pFirstSon;
709 for(; son; son = son->m_pBrother) {
710
711 switch(id(son)) {
712 case nodePredefKey: {
713 if (son->m_valueType != xmlListBegin) break;
714
715 // set attributes to default values
716 int vId = idCount++;
717
718 // create new node if necessary and assign attributes
719 if (mapToNode[vId] == 0)
720 mapToNode[vId] = G.newNode();
721 break;
722 }
723 case edgePredefKey:
724 if (son->m_valueType != xmlListBegin) break;
725
726 // set attributes to default values
727 int sourceId = notDefined, targetId = notDefined;
728 // read all relevant attributes
729 XmlObject *edgeSon = son->m_pFirstSon;
730 for(; edgeSon; edgeSon = edgeSon->m_pBrother) {
731 int i = 0;
732 switch(id(edgeSon)) {
733 case sourcePredefKey:
734 if (edgeSon->m_valueType != xmlStringValue) break;
735 for (i = 0; i<= maxId;i++)
736 if (!strcmp(idMap[i],edgeSon->m_stringValue))
737 sourceId = i;
738 break;
739
740 case targetPredefKey:
741 if (edgeSon->m_valueType != xmlStringValue) break;
742 for (i = 0; i<= maxId;i++)
743 if (!strcmp(idMap[i],edgeSon->m_stringValue))
744 targetId = i;
745 break;
746
747 }
748 }
749
750 // check if everything required is defined correctly
751 if (sourceId == notDefined || targetId == notDefined) {
752 setError("source or target id not defined");
753 //cout << "source or target id not defined" << endl;
754 return false;
755
756 } else if (sourceId < minId || maxId < sourceId ||
757 targetId < minId || maxId < targetId) {
758 setError("source or target id out of range");
759 //cout << "source or target id out of range" << endl;
760 return false;
761 }
762
763 // create adjacent nodes if necessary and new edge
764 if (mapToNode[sourceId] == 0) mapToNode[sourceId] = G.newNode();
765 if (mapToNode[targetId] == 0) mapToNode[targetId] = G.newNode();
766
767
768 G.newEdge(mapToNode[sourceId],mapToNode[targetId]);
769 break;
770 }
771 }
772
773 return true;
774}
