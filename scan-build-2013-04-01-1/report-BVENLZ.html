725 itC = cList.rbegin().succ();
726 }//while
727
728 //**********
729 //set status
730 OGDF_ASSERT(arcTwin[eArc])
731 if ((ind1 > 0 ) && (ind2 > 0))
732 {
733 if (ind1 == ind2) //bidirectional
734 {
735 m_eStatus[eArc] = 1;
736 m_eStatus[arcTwin[eArc]] = 1;
737 }//if both
738 else
739 if (ind1 < ind2)
740 {
741 m_eStatus[eArc] = 1;
742 m_eStatus[arcTwin[eArc]] = 0;
743 }
744 else
745 {
746 m_eStatus[eArc] = 0;
747 m_eStatus[arcTwin[eArc]] = 1;
748 }
749
750 }//if
751 else
752 {
753 //remove edge
754 m_eStatus[eArc] = 0;
755 m_eStatus[arcTwin[eArc]] = 0;
756 }//else
757}//setArcStatus
